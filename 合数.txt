//（4）合数就是非素数，即除了1和它本身之外还有其他约数的正整数。
//编写一个程序求出指定数据范围（假设10~100）内的所有合数。
namespace exercise4
{
    class Program
    {
        static void Main(string[] args)
        {
            for(int i =10;i<=100;i++)
                for(int j =2;j<i/2;j++)
                    if (i % j == 0)
                    {
                        Console.Write(i);
                        Console.Write("   ");
                        break;
                    }
        }
    }
}
